set -u
set -o pipefail

# $snippet my_nginx_app_wo_sidecar_mtls_disabled_apply syntax="bash" outputis="text"
$ kubectl apply -f <(istioctl kube-inject -f @samples/https/nginx-app.yaml@)
# $verify
service/my-nginx created
replicationcontroller/my-nginx created
# $endsnippet

# $snippet sleep_app_wo_sidecar_mtls_disabled_apply syntax="bash" outputis="text"
$ kubectl apply -f <(istioctl kube-inject -f @samples/sleep/sleep.yaml@)
# $verify
serviceaccount/sleep created
service/sleep created
deployment.apps/sleep created
# $endsnippet

kubectl wait --for=condition=available deployment --all --timeout=300s
kubectl wait --for=condition=Ready pod --all --timeout=300s

# $snippet create_configmap_wo_sidecar_mtls_disabled syntax="bash"
$ kubectl get pod
# $verify verifier="lineRegex"
NAME *READY *STATUS *RESTARTS *AGE
my-nginx-[0-9a-z]* *1/1 *Running *0
sleep-[0-9a-z]*-[0-9a-z]* *2/2 *Running *0
# $endsnippet


# $snippet welcome_to_nginx_wo_sidecar_mtls_disabled outputis="text"
$ kubectl exec $(kubectl get pod -l app=sleep -o jsonpath={.items..metadata.name}) -c istio-proxy -- curl -s https://my-nginx -k
# $verify verifier="contains"
<h1>Welcome to nginx!</h1>
# $endsnippet
